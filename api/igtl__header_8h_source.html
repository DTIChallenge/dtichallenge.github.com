48 /*
49  * Message header of OpenIGTLink message.
50  *
51  * igtl_header is an overall data header for OpenIGTLink protocol.
52  * It is transfered at beginning of every OpenIGTLink message to give
53  * type and size of following data body to a receiver.
54  * These parameters allow the receiver to parse or skip the data body.
55  */
56 
57 typedef struct {
58  igtl_uint16 version; /* protocol version number */
59  char name[IGTL_HEADER_TYPE_SIZE]; /* data type name */
60  char device_name[IGTL_HEADER_NAME_SIZE]; /* device name */
61  igtl_uint64 timestamp; /* time stamp message */
62  igtl_uint64 body_size; /* size of the body */
63  igtl_uint64 crc; /* CRC */
64 } igtl_header;
