51 /*
52  * Image meta data OpenIGTLinik protocol
53  *
54  * IMAGEMETA is a message type to transfer a list of images available in a server.
55  * A IMGMETA message may contain information of multiple images.
56  * The client determins the number of image meta by the size of the body included
57  * in the message header (see igtl_imgmeta_getdata_n() macro).
58  */
59 
60 typedef struct {
61  char name[IGTL_IMGMETA_LEN_NAME]; /* name / description */
62  char device_name[IGTL_IMGMETA_LEN_DEVICE_NAME]; /* device name to query the IMAGE and COLORT */
63  char modality[IGTL_IMGMETA_LEN_MODALITY]; /* modality name */
64  char patient_name[IGTL_IMGMETA_LEN_PATIENT_NAME]; /* patient name */
65  char patient_id[IGTL_IMGMETA_LEN_PATIENT_ID]; /* patient ID (MRN etc.) */
66  igtl_uint64 timestamp; /* scan time */
67  igtl_uint16 size[3]; /* entire image volume size */
68  igtl_uint8 scalar_type; /* scalar type. see scalar_type in IMAGE message */
69  igtl_uint8 reserved;
70 } igtl_imgmeta_element;

75 /*
76  * Macros for image meta data size
77  *
78  * igtl_imgmeta_get_data_size(n) calculates the size of body based on the number
79  * of images.The size of body is used in the message header.
80  * igtl_imgmeta_get_data_n(size) calculates the number of images in the body, based on
81  * the body size. This function may be used when a client program parses IMGMETA message.
82  *
83  */
84 
85 #define igtl_imgmeta_get_data_size(n) ((n) * IGTL_IMGMETA_ELEMENT_SIZE)
86 #define igtl_imgmeta_get_data_n(size) ((size) / IGTL_IMGMETA_ELEMENT_SIZE)
