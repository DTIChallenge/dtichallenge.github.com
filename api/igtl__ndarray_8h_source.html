44 /*
45  * NDARRAY is a data type, which is designed to transfer N-dimensional numerical array.
46  * The message body consists of N-D array header, size table, and N-D array body.
47  */
48 
49 typedef struct {
50  igtl_uint8 type; /* Scalar type (2:int8 3:uint8 4:int16 5:uint16 6:int32
51  7:uint32 10:float32 11:float64 13:complex) */
52  igtl_uint8 dim; /* Dimension of array */
53  igtl_uint16 * size; /* Array size */
54  void * array;
55 } igtl_ndarray_info;
