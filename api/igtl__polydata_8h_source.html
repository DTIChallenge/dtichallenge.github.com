62 typedef struct {
63  igtl_uint8 type; /* attribute type */
64  /* Values for TYPE_ATTRIBUTE (16-bit)
65  0x00: POINT_DATA / Scalars
66  0x01: POINT_DATA / Vectors
67  0x02: POINT_DATA / Normals
68  0x03: POINT_DATA / Tensors
69  0x10: CELL_DATA / Scalars

71  0x12: CELL_DATA / Normals
72  0x13: CELL_DATA Tensors */
73  igtl_uint8 ncomponents; /* number of components */
74  /* must be 3 for Vectors and Normal, 9 for Tensor.*/
