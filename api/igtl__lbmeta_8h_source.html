32 /*
33  * Label meta data in OpenIGTLinik protocol
34  *
35  * LBMETA is a data type to send a list of label image data available in a server.
36  * In OpenIGTLink protocol, actual label data is transfered as an IMAGE message.
37  * LBMETA message lets the receiving client know which IMAGE data are available
38  * as label maps.
39  * Multiple labels in a LBMETA message can point the same IMAGE message, if there
40  * are multiple lables are recoreded in a single IMAGE mesage. For example, given
41  * the following label map as IMAGE data with name "LableImage1":
42  *
43  * 0 0 0 1 0 0 value | name / description
44  * 2 0 1 1 1 0 --------+----------------------------
45  * 2 2 1 1 1 0 0 | None
46  * 2 2 2 1 0 0 1 | Liver
47  * 2 2 2 2 0 0 2 | Kidney
48  *
49  * To send the label information recorded in "LabelImage1," one can create
50  * a following LBMETA message:
51  *
52  * DATA | Contents
53  * ---------------+---------------------------
54  * NAME 1 | Liver
55  * IMAGE NAME 1 | LabelImage
56  * Labe l | 1
57  * RGBA 1 | 0xFF0000
58  * SIZE 1 | (6, 5, 1)
59  * OWNER 1 |
60  * --------------+---------------------------
61  * NAME 2 | Kidney
62  * IMAGE NAME 2 | LabelImage
63  * Labe 2 | 2
64  * RGBA 2 | 0xFF2222
65  * SIZE 2 | (6, 5, 1)
66  * OWNER 2 |
67  * --------------+---------------------------
68  *
69  * ... ....
70  *
71  * --------------+---------------------------
72  *
73  * The client determins the number of image meta by the size of the body included
74  * in the message header (see igtl_lbmeta_getdata_n() macro).
75  */
76 
77 #pragma pack(1) /* For 1-byte boundary in memroy */

79 typedef struct {
80  char name[IGTL_LBMETA_LEN_NAME]; /* name / description */
81  char device_name[IGTL_LBMETA_LEN_DEVICE_NAME]; /* device name to query the IMAGE */
82  igtl_uint8 label; /* label */
83  igtl_uint8 reserved;
84  igtl_uint8 rgba[4]; /* Color in RGBA. default: (0, 0, 0, 0) */
85  igtl_uint16 size[3]; /* Number of pixels in each direction */
86  char owner[IGTL_LBMETA_LEN_OWNER];/* Device name of the owner image. (can be empty) */
87 } igtl_lbmeta_element;

92 /*
93  * Macros for label meta data size
94  *
95  * igtl_lbmeta_get_data_size(n) calculates the size of body based on the number
96  * of images.The size of body is used in the message header.
97  * igtl_lbmeta_get_data_n(size) calculates the number of images in the body, based on
98  * the body size. This function may be used when a client program parses LBMETA message.
99  *
100  */
101 
102 #define igtl_lbmeta_get_data_size(n) ((n) * IGTL_LBMETA_ELEMENT_SIZE)
103 #define igtl_lbmeta_get_data_n(size) ((size) / IGTL_LBMETA_ELEMENT_SIZE)
